<div class='form-horizontal well'>
   <fieldset>
      <legend>
         Find Employees By:
      </legend>
      <div ng-repeat="filterOption in filterOptions">
         <label ng-if="filterOption.fieldName=='First_Name'" class="col-md-6">
         First Name:
         <input ng-model="filterOption.value" />
         </label>
         <label ng-if="filterOption.fieldName=='Job_Title'" class="col-md-6">
            Job Title:
            <select ng-model="filterOption.value">
               <option ng-repeat="Job_TitleOption in Job_TitleOptions" value="{{Job_TitleOption.Id}}">
                  {{Job_TitleOption.Name}}
               </option>
            </select>
         </label>
      </div>
   </fieldset>
</div>
<bknd-ng-grid view-name="viewName" input-style="inputStyle" filter-options="filterOptions"></bknd-ng-grid>
backAndControllers.controller('filterController', ['$scope', 'dataListService', function ($scope, dataListService)  {
    $scope.viewName  = 'Employees';
    $scope.inputStyle  =  {
        'height': 300
    };
    $scope.buttonGroups  = [ {
        buttons: [ {
            text: "New"
            , iconClass: "glyphicon-plus"
            , callback: function ()  {
                alert("New")
            }
        }
        ,  {
            text: "Edit"
            , iconClass: "glyphicon-pencil"
            , callback: function ()  {
                alert("Edit")
            }
        }
        ,  {
            text: "Delete"
            , iconClass: "glyphicon-trash"
            , callback: function ()  {
                alert("Delete")
            }
        }
        ]
    }
    ];
    $scope.filterOptions  = [ {
        fieldName: 'First_Name'
        , operator: 'contains'
        , value: ''
    }
    ,  {
        fieldName: 'Job_Title'
        , operator: 'in'
        , value: ''
    }
    ];
    $scope.loadSelectOptions  = function ()  {
        dataListService.read({
            dataType: "view"
            , viewName: "EmployeesJob_Title"
            , withSelectOptions: null
            , filter: null
            , sort: JSON.stringify([ {
                fieldName:"Ordinal"
                , order: "asc"
            }
            ])
            , search: null
        }
        , function (data)  {
            $scope.Job_TitleOptions  = data.data;
        });
    }
    $scope.loadSelectOptions();
}
])
